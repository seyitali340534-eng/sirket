rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Requests collection: allow create from public, read only to admins (by email)
    match /requests/{docId} {
      allow create: if true;           // herkes form gönderebilir
      allow read, update, delete: if request.auth != null
        && request.auth.token.email.matches('.*@.*'); // basit geliştirme şartı
    }
  }
}
